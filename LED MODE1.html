LED:MODE1

依據環境亮暗程度，自動切換RGB LED燈的色溫。


利用光敏電阻感應周遭環境的亮暗程度，

採呼吸燈的方式逐漸改變顏色，

亮度較高時為冷色系，亮度較低時為暖色系。


部 分程式碼如下：

while(inB == '5'){
      inB2 = inB;
      val=analogRead(A2);
      if(val>800)
      {//環境較亮，冷色燈
            if (Serial.available() > 0) inB2 = Serial.read();
            if(inB != inB2) {
              inB = inB2;
              break;      
            }
            analogWrite(LEDR, 102);
            analogWrite(LEDG, 50);
            analogWrite(LEDB, 200);
      }
      else
      {//環境較暗，暖色系
          /*turnoff();*/
          for(int i=0;i<150;i++)
          {
            if (Serial.available() > 0) inB2 = Serial.read();
            if(inB != inB2) {
              inB = inB2;
              break;      
            }
            val=analogRead(A2);
            Serial.println(val);
            if(val>800) break;
            analogWrite(LEDR, 255);
            analogWrite(LEDG, i);
            analogWrite(LEDB, 0);
            delay(30);
          }
         for(int i=149;i>-1;i--)
          {
            if (Serial.available() > 0) inB2 = Serial.read();
            if(inB != inB2) {
              inB = inB2;
              break;      
            }
            val=analogRead(A2);
            Serial.println(val);
            if(val>800) break;
            analogWrite(LEDR, 255);
            analogWrite(LEDG, i);
            analogWrite(LEDB, 0);
            delay(30);
          }
      }
    }
